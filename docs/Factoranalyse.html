cor_with_stars <- function(cor_matrix, p_matrix) {
    stars <- ifelse(is.na(p_matrix), "", ifelse(p_matrix < 0.001, "***", ifelse(p_matrix < 0.01, "**",
        ifelse(p_matrix < 0.05, "*", ""))))

    cor_matrix_formatted <- matrix(paste0(format(cor_matrix, nsmall = 3), stars), nrow = nrow(cor_matrix),
        dimnames = dimnames(cor_matrix))

    return(cor_matrix_formatted)
}
