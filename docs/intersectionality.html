Intersectionality

NEA2022 <- NEA2022 %>%
    mutate(gender = recode(m_v, man = "m", vrouw = "v"))


NEA2022 <- NEA2022 %>%
    mutate(migration = recode(vrllandd, Nederland = "nl", `Europa (exclusief Nederland)` = "eu", `Buiten Europa` = "noeu"))

NEA2022 <- NEA2022 %>%
    mutate(education = recode(afl_soi3, `laag opleidingsniveau` = "laag", `middelbaar opleidingsniveau` = "middel",
        `hoog opleidingsniveau` = "hoog"))

NEA2022$intersections <- interaction(NEA2022$gender, NEA2022$migration, NEA2022$education, NEA2022$leeftijd_)
